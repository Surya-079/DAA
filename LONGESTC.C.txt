#include<stdio.h>
#include<string.h>
#include<time.h>
int max(int a,int b){
return(a>b)?a:b;
}
int lcs(char *X,char *Y,int m,int n){
int L[100][100];
char lcs_str[100];
int index;
int i,j;
int k,l;
for(i=0;i<=m;i++){
for(j=0;j<=n;j++){
if(i==0 || j==0){
L[i][j]=0;
}else if(X[i-1]==Y[j-1]){
L[i][j]=L[i-1][j-1]+1;
}else{
L[i][j]=max(L[i-1][j],L[i][j-1]);
}
}
}
index=L[m][n];
lcs_str[index]='\0';
k=m;
l=n;
while(k>0 && l>0){
if(X[k-1]==Y[l-1]){
lcs_str[index-1]=X[k-1];
k--;
l--;
index--;
}else if(L[k-1][l]>L[k][l-1]){
k--;
}else{
l--;
}
}
printf("Longest Common Subsequence: %s\n",lcs_str);
return L[m][n];
}
int main(){
int m,n;
char X[]="AGGTAB";
char Y[]="GXTXAYB";
clock_t start,end;
double total_time;
clrscr();
m=strlen(X);
n=strlen(Y);
start=clock();
printf("Length of LCS is %d\n",lcs(X,Y,m,n));
end=clock();
total_time=((double)(end-start))/CLOCKS_PER_SEC;
printf("Execution Time: %f seconds",total_time);
getch();
return 0;
}
